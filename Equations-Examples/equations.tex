\section{Definitions of Ses}
\begin{enumerate}
    \item As a limit:
          \[e= \lim_{n\to\infty} \left(1+\frac{1}{n}\right)^n\]
    \item As a sum:
          \[e=\sum_{n=0}^{\infty} \frac{1}{n!}.\]
    \item As a continued fraction:
          \[e=2+\frac{1}{1+\frac{1}{2+\frac{2}{3+\frac{3}{4+\frac{4}{5+\ddots}}}}}\]
\end{enumerate}
